#ifndef _NGINX_H_INCLUDED_
#define _NGINX_H_INCLUDED_


#define nginx_version      1004004
#define NGINX_VERSION      ""
#define NGINX_VER          "" NGINX_VERSION

#define NGINX_VAR          ""
